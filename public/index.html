const express = require('express'); const app = express(); const { Configuration, OpenAIApi } = require('openai'); app.use(express.json()); const configuration = new Configuration({ apiKey: process.env.OPENAI_API_KEY, }); const openai = new OpenAIApi(configuration); app.post('/chat', async (req, res) => { const { message, prompt, tema } = req.body; const finalPrompt = `${prompt} El tema actual es: ${tema}. Asegurate de mantenerte en contexto.`; try { const chatCompletion = await openai.chat.completions.create({ model: 'gpt-4o', messages: [ { role: 'system', content: finalPrompt }, { role: 'user', content: message } ] }); res.json({ reply: chatCompletion.choices[0].message.content }); } catch (err) { console.error("ERROR:", err); res.status(500).send('Error al comunicarse con OpenAI'); } }); app.listen(3000, () => { console.log('Server running '); });
